a)
\begin{align*}
  P(A \cup B) &= P(A) + P(B) - P(A,B) \Leftrightarrow \\
  P(A,B) &= P(A) + P(B) - P(A \cup B) \\
         &= 0.4 + 0.7 - 0.9 = 0.2 \\
\end{align*}

b)
\begin{align*}
  P(\bar{A}, B) = P(B-A) = P(B) - P(A,B) = 0.7 - 0.2 = 0.5
\end{align*}

c)
\begin{align*}
  P(A-B) &= P(A) - P(A,B) \\
         &= P(A \cup B) - P(B) \\
         &= 0.9 - 0.7 = 0.2
\end{align*}

d)
\begin{align*}
  P(\bar{A} - B) &= P(S - A - B) \\
                 &= 1 - P(A) - P(B) + P(A,B) \\
                 &= 1 - 0.4 - 0.7 + 0.2 = 0.1
\end{align*}

with $S$ denoting the sample space

e)
\begin{align*}
  P(\bar{A} \cup B) &= P((S-A) \cup B) \\
                    &= P(S-A) + P(B) - P((S-A),B) \\
                    &= 0.6 + 0.7 - 0.5 = 0.8
\end{align*}

with $S$ denoting the sample space

f)
\begin{align*}
  P(A, (\bar{A} \cup B)) &= P(A) + P(\bar{A} \cup B) - P(A \cup (\bar{A} \cup B)) \\
                         &= P(A) + P(\bar{A} \cup B) - P(S) \\
                         &= 0.4 + 0.8 - 1 = 0.2
\end{align*}

with $S$ denoting the sample space
