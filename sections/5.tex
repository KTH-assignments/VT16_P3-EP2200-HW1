\begin{align*}
  E[Y] &= E[-2X + 3] = -2E[X] + 3 \Leftrightarrow \\
     1 &= -2E[X] + 3 \Leftrightarrow \\
  E[X] &= 1
\end{align*}

and

\begin{align*}
  E[Y^2] &= E[(-2X + 3)^2] = E[4X^2 - 12X + 9] \Leftrightarrow \\
       9 &= 4E[X^2] - 12E[X] + 9 \Leftrightarrow \\
  E[X^2] &= 3E[X] = 3
\end{align*}

Hence

\begin{align*}
  Var(X) = E[X^2] - E^2[X] = 3 - 1 = 2
\end{align*}
